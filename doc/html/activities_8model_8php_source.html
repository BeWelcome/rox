117  $longsw = rad2deg((6378 * $long - $distance) / 6378);

120  $latne = rad2deg(($distance + $radiusAtLatitude * $lat) / $radiusAtLatitude);
121  $latsw = rad2deg(($radiusAtLatitude * $lat - $distance) / $radiusAtLatitude);
122  if ($latne < $latsw) {

124  $latne = $latsw;
125  $latsw = $tmp;
126  }
