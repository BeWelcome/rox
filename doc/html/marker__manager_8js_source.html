20  * Marker manager is an interface between the map and the user, designed
21  * to manage adding and removing many points when the viewport changes.
22  *
23  *
24  * Algorithm: The MM places its markers onto a grid, similar to the map tiles.
25  * When the user moves the viewport, the MM computes which grid cells have
26  * entered or left the viewport, and shows or hides all the markers in those
27  * cells.
28  * (If the users scrolls the viewport beyond the markers that are loaded,
29  * no markers will be visible until the EVENT_moveend triggers an update.)
30  *
31  * In practical consequences, this allows 10,000 markers to be distributed over
32  * a large area, and as long as only 100-200 are visible in any given viewport,
33  * the user will see good performance corresponding to the 100 visible markers,
34  * rather than poor performance corresponding to the total 10,000 markers.
35  *
36  * Note that some code is optimized for speed over space,
37  * with the goal of accommodating thousands of markers.
38  *
39  */

171  var mPoint = marker.getPoint();
172  // Tracking markers is expensive, so we do this only if the
173  // user explicitly requested it when creating marker manager.
174  if (this.trackMarkers_) {
175  GEvent.bind(marker, "changed", this, this.onMarkerMoved_);

199 MarkerManager.prototype.isGridPointVisible_ = function(point) {
200  var me = this;
201  var vertical = me.shownBounds_.minY <= point.y &&
202  point.y <= me.shownBounds_.maxY;
203  var minX = me.shownBounds_.minX;
204  var horizontal = minX <= point.x && point.x <= me.shownBounds_.maxX;
205  if (!horizontal && minX < 0) {
206  // Shifts the negative part of the rectangle. As point.x is always less
207  // than grid width, only test shifted minX .. 0 part of the shown bounds.
208  var width = me.gridWidth_[me.shownBounds_.z];
209  horizontal = minX + width <= point.x && point.x <= width - 1;
210  }
211  return vertical && horizontal;
212 }

224 MarkerManager.prototype.onMarkerMoved_ = function(marker, oldPoint, newPoint) {
225  // NOTE: We do not know the minimum or maximum zoom the marker was
226  // added at, so we start at the absolute maximum. Whenever we successfully
227  // remove a marker at a given zoom, we add it at the new grid coordinates.
228  var me = this;
229  var zoom = me.maxZoom_;
230  var changed = false;
231  var oldGrid = me.getTilePoint_(oldPoint, zoom, GSize.ZERO);
232  var newGrid = me.getTilePoint_(newPoint, zoom, GSize.ZERO);
233  while (zoom >= 0 && (oldGrid.x != newGrid.x || oldGrid.y != newGrid.y)) {
234  var cell = me.getGridCellNoCreate_(oldGrid.x, oldGrid.y, zoom);
235  if (cell) {
236  if (me.removeFromArray(cell, marker)) {
237  me.getGridCellCreate_(newGrid.x, newGrid.y, zoom).push(marker);
238  }
239  }
240  // For the current zoom we also need to update the map. Markers that no
241  // longer are visible are removed from the map. Markers that moved into
242  // the shown bounds are added to the map. This also lets us keep the count
243  // of visible markers up to date.
244  if (zoom == me.mapZoom_) {
245  if (me.isGridPointVisible_(oldGrid)) {
246  if (!me.isGridPointVisible_(newGrid)) {
247  me.removeOverlay_(marker);
248  changed = true;
249  }
250  } else {
251  if (me.isGridPointVisible_(newGrid)) {
252  me.addOverlay_(marker);
253  changed = true;
254  }
255  }
256  }
257  oldGrid.x = oldGrid.x >> 1;
258  oldGrid.y = oldGrid.y >> 1;
259  newGrid.x = newGrid.x >> 1;
260  newGrid.y = newGrid.y >> 1;
261  --zoom;
262  }
263  if (changed) {
264  me.notifyListeners_();
265  }
266 };

444 MarkerManager.prototype.getGridBounds_ = function(bounds, zoom, swPadding,
445  nePadding) {
446  zoom = Math.min(zoom, this.maxZoom_);
447 
448  var bl = bounds.getSouthWest();
449  var tr = bounds.getNorthEast();
450  var sw = this.getTilePoint_(bl, zoom, swPadding);
451  var ne = this.getTilePoint_(tr, zoom, nePadding);
452  var gw = this.gridWidth_[zoom];
453 
454  // Crossing the prime meridian requires correction of bounds.
455  if (tr.lng() < bl.lng() || ne.x < sw.x) {
456  sw.x -= gw;
457  }
458  if (ne.x - sw.x + 1 >= gw) {
459  // Computed grid bounds are larger than the world; truncate.
460  sw.x = 0;
461  ne.x = gw - 1;
462  }
463  var gridBounds = new GBounds([sw, ne]);
464  gridBounds.z = zoom;
465  return gridBounds;
466 };

535  var me = this;
536  me.mapZoom_ = this.map_.getZoom();
537  var newBounds = me.getMapGridBounds_();
538 
539  // If the move does not include new grid sections,
540  // we have no work to do:
541  if (newBounds.equals(me.shownBounds_) && newBounds.z == me.shownBounds_.z) {
542  return;
543  }
544 
545  if (newBounds.z != me.shownBounds_.z) {
546  me.processAll_(me.shownBounds_, me.removeOverlay_);
547  me.processAll_(newBounds, me.addOverlay_);

551

654 MarkerManager.prototype.rectangleDiffCoords = function(bounds1, bounds2, callback) {
655  var minX1 = bounds1.minX;
656  var minY1 = bounds1.minY;
657  var maxX1 = bounds1.maxX;
658  var maxY1 = bounds1.maxY;
659  var minX2 = bounds2.minX;
660  var minY2 = bounds2.minY;
661  var maxX2 = bounds2.maxX;
662  var maxY2 = bounds2.maxY;
663 
664  for (var x = minX1; x <= maxX1; x++) { // All x in R1
665  // All above:
666  for (var y = minY1; y <= maxY1 && y < minY2; y++) { // y in R1 above R2
667  callback(x, y);
668  }
669  // All below:
670  for (var y = Math.max(maxY2 + 1, minY1); // y in R1 below R2
671  y <= maxY1; y++) {
672  callback(x, y);
673  }
674  }
675 
676  for (var y = Math.max(minY1, minY2);
677  y <= Math.min(maxY1, maxY2); y++) { // All y in R2 and in R1
678  // Strictly left:
679  for (var x = Math.min(maxX1 + 1, minX2) - 1;
680  x >= minX1; x--) { // x in R1 left of R2
681  callback(x, y);
682  }
683  // Strictly right:
684  for (var x = Math.max(minX1, maxX2 + 1); // x in R1 right of R2
685  x <= maxX1; x++) {
686  callback(x, y);
687  }
688  }
689 };
