59  public function index($args = false)
60  {
61  if (PPostHandler::isHandling()) {
62  return;
63  }
64 
65  $request = $args->request;
66 
67  if (isset($request[0]) && 'rox' == $request[0]) {
68  // bw.org/rox/in/lang or bw.org/rox/start
69  // should be the same as just
70  // bw.org/in/lang, or bw.org/start
71  array_shift($request);
72  }
73  switch (isset($request[0]) ? $request[0] : false) {
74  case 'in':
75  // language switching
76  if (!isset($request[1])) {
77  $this->redirectHome();
78  } else {
79  $this->_switchLang($request[1]);
80  $this->redirect(array_slice($request, 2), $args->get);
81  }
82  PPHP::PExit();
83  case 'trmode': // an alias..
84  case 'tr_mode':
85  // translation mode switching
86  if (!isset($request[1])) {
87  $this->redirectHome();
88  } else {
89  $this->_switchTrMode($request[1]);
90  $this->redirect(array_slice($request, 2), $args->get);
91  }
92  PPHP::PExit();
93  case 'start':
94  $page = new PublicStartpage();
95  break;
96  case 'trac':
97  case 'mediawiki':
98  case 'mailman':
99  $this->redirectAbsolute('http://www.bevolunteer.org/'.$request[0]);
100  PPHP::PExit();
101  case 'www.bewelcome.org':
102  // some emails sent by mailbot contain a link to
103  // http://www.bewelcome.org/www.bewelcome.org/something
104  // we need to redirect them to
105  // https://www.bewelcome.org/something
106  $this->redirect(array_slice($request, 1), $args->get);
107  PPHP::PExit();
108  case 'main':
109  case 'home':
110  case 'index':
111  case '':
112  default:
113  if (APP_User::isBWLoggedIn("NeedMore,Pending")) {
114  $page = new PersonalStartpage(); // This is the Main Start page for logged in members
115  } else {
116  $page = new PublicStartpage(); // This is the Default Start page for not logged in members
117  }
118  }
119 
120  $page->setModel($this->_model);
121  $page->model = $this->_model; // some want it like this
122 
123  return $page;
124  }
