38  public function passthroughCSS($req)
39  {
40  $loc = PApps::getBuildDir().'rox/'.$req;
41  if (!file_exists($loc))
42  exit();
43  $headers = apache_request_headers();
44  // Checking if the client is validating his cache and if it is current.
45  if (isset($headers['If-Modified-Since']) && (strtotime($headers['If-Modified-Since']) == filemtime($loc))) {
46  // Client's cache IS current, so we just respond '304 Not Modified'.
47  header('Last-Modified: '.gmdate('D, d M Y H:i:s', filemtime($loc)).' GMT', true, 304);
48  } else {
49  // File not cached or cache outdated, we respond '200 OK' and output the image.
50  header('Last-Modified: '.gmdate('D, d M Y H:i:s', filemtime($loc)).' GMT', true, 200);
51  header('Content-Length: '.filesize($loc));
52  }
53  header('Content-type: text/css');
54  @copy($loc, 'php://output');
55  exit();
56  }
