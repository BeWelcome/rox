80  private function fage_value($dd) {
81  $pieces = explode("-",$dd);
82  if(count($pieces) != 3) return 0;
83  list($year,$month,$day) = $pieces;
84  $year_diff = date("Y") - $year;
85  $month_diff = date("m") - $month;
86  $day_diff = date("d") - $day;
87  if ($month_diff < 0) $year_diff--;
88  elseif (($month_diff==0) && ($day_diff < 0)) $year_diff--;
89  return $year_diff;
90  } // end of fage_value

192  private function locationWhere($vars, $admin1, $country) {
193  if ($country) {
194  if ($admin1) {
195  // We run based on an admin unit
196  $where = "AND a.IdCity = g.geonameid
197  AND g.admin1 = '" . $admin1 . "'
198  AND g.country = '" . $country . "'";
199  } else {
200  // we're looking for all members of a country
201  $where = "AND a.IdCity = g.geonameid
202  AND g.country = '" . $country . "'";
203  }
204  } else {
205  $where = "AND a.IdCity = g.geonameid";
206  if (empty($vars['search-location'])) {
207  // we search around the world.
208  } else {
209  // a simple place with a square rectangle around it
210  $distance = $vars['search-distance'];
211  if ($distance != 0) {
212  // calculate rectangle around place with given distance
213  $lat = deg2rad(doubleval($vars['search-latitude']));
214  $long = deg2rad(doubleval($vars['search-longitude']));
215 
216  $longne = rad2deg(($distance + self::EARTH_RADIUS * $long) / self::EARTH_RADIUS);
217  $longsw = rad2deg((self::EARTH_RADIUS * $long - $distance) / self::EARTH_RADIUS);
218 
219  $radiusAtLatitude = cos($lat) * self::EARTH_RADIUS;
220  $latne = rad2deg(($distance + $radiusAtLatitude * $lat) / $radiusAtLatitude);
221  $latsw = rad2deg(($radiusAtLatitude * $lat - $distance) / $radiusAtLatitude);
222  // Sanity check if $latne < $latsw or $longne < $longsw switch the two (Melbourne)
223  // TODO: search around the date line
224  if ($latne < $latsw) {
225  $tmp = $latne;
226  $latne = $latsw;
227  $latsw = $tmp;
228  }
229  if ($longne < $longsw) {
230  $tmp = $longne;
231  $longne = $longsw;
232  $longsw = $tmp;
233  }
234  // now fetch all location from geonames which are in that given rectangle
235  $query = "
236  SELECT
237  g.geonameid AS geonameid
238  FROM
239  geonames g
240  WHERE
241  " . self::PLACES_FILTER . "
242  AND g.latitude BETWEEN " . $latsw . " AND " . $latne . "
243  AND g.longitude BETWEEN " . $longsw . " AND " . $longne;
244  $where .= "
245  AND g.geonameid IN ('";
246  $geonameids = $this->bulkLookup($query);
247  foreach($geonameids as $geonameid) {
248  $where .= $geonameid->geonameid . "', '";
249  }
250  $where = substr($where, 0, -3) . ")";
251  } else {
252  $where .= " AND g.geonameid = " . $this->dao->escape($vars['search-geoname-id']);
253  }
254  }
255  }
256  return $where;
257  }
