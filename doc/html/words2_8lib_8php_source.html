70  function translate($code, $context, $args)
71  {
72  $first_language = false;
73  foreach ($this->spoken_languages as $lang) {
74  $translation = $this->words_gateway->getTranslation($code, $lang->id);
75  if (is_object($translation)) {
76  $text = sprintf($translation->Sentence, $args);
77  if (!$first_language) {
78  // TODO: check for obsolete translations!
79  if (false) {
80  return $this->print_strategy_map['obsolete']->$context($text, $code, $lang, 'obsolete');
81  }
82  return $this->print_strategy_map['successful']->$context($text, $code, $lang, 'successful');
83  }
84  // translation found, but not in original language...
85  return $this->print_strategy_map['missing_translation']->$context($text, $code, $first_language, 'missing_translation');
86  }
87  $first_language = $lang;
88  }
89  // no translation found..
90  // we assume that the last in the array of $spoken_languages is always the reference language..
91  return $this->print_strategy_map['missing_word']->$context($code, $code, $lang, 'missing_word');
92  }
