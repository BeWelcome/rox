532  {
533  if (! $this->_offerTranslationLink) {

542  case LookedUpWord::MISSING_TR:

547  }
548  else {
549  return $word->clickableText();

580  private function _lookup($code, $args, $lang = false, $get_raw = false) {
581  if($lang == false) {
582  $lang = $this->_lang;
583  }
584 
585  $row = $this->_lookup_row($code, $lang);
586  // select the English wordcode only if not archived
587  $row_en = $this->_lookup_row($code, 'en','AND (isarchived=0 or isarchived is null)');
588  $R = MOD_right::get();
589 
590  if(! $this->_offerTranslationLink) {
591  // for normal people no translation stuff
592  if(!$row_en) {
593  // if English doesn't exist, show code
594  $lookup_result = $code;
595  } elseif (!$row || $row->updated < $row_en->majorupdate) {
596  // if no translation or translation is outdated, show English
597  $lookup_result = $this->_modified_sentence_from_row($row_en, $args, $get_raw);
598  } else {
599  // if up-to-date translation present, show translation
600  $lookup_result = $this->_modified_sentence_from_row($row, $args, $get_raw);
601  }
602 
603  return new LookedUpWord($code, $lang, $lookup_result);
604  } else {
605  // for translators, the LookedUpWord object needs more info
606  $tr_quality = LookedUpWord::FINE;
607  if (!$row_en){
608  // there is no English text: show the wordcode
609  // show as missing if user has enough rights
610  if ($R->hasRight("Words")<10){
611  $tr_success = LookedUpWord::NO_TR_LINK;
612  } else {
613  $tr_success = LookedUpWord::MISSING_WORD;
614  }
615  $lookup_result = $code;
616  } elseif (!$row){
617  // there is no translation: show the English text and as missing
618  $tr_success = LookedUpWord::MISSING_TR;
619  $lookup_result = $this->_modified_sentence_from_row($row_en, $args, $get_raw);
620  } elseif (!($row_en->updated > $row->updated)){
621  // the English text has not been updated at all after the translation: show translation normally
622  $tr_success = LookedUpWord::SUCCESSFUL;
623  $lookup_result = $this->_modified_sentence_from_row($row, $args, $get_raw);
624  } elseif (!($row_en->majorupdate > $row->updated)){
625  // the English text did not have a major update after the translation: show translation and as obsolete
626  $tr_success = LookedUpWord::OBSOLETE;
627  $lookup_result = $this->_modified_sentence_from_row($row, $args, $get_raw);
628  } else {
629  // the English text did have a major update after the translation: show English text and as obsolete
630  $tr_success = LookedUpWord::OBSOLETE;
631  $lookup_result = $this->_modified_sentence_from_row($row_en, $args, $get_raw);
632  }
633 
634  if (!$row_en) {
635  } else {
636  if ($row_en->donottranslate == 'yes') {
637  // text is dnt: show normally
638  $tr_success = LookedUpWord::NO_TR_LINK;
639  if (!$row || $row->updated < $row_en->majorupdate) {
640  $lookup_result = $this->_modified_sentence_from_row($row_en, $args, $get_raw);
641  } else {
642  $lookup_result = $this->_modified_sentence_from_row($row, $args, $get_raw);
643  }
644  }
645  }
646 
647  switch ($this->_trMode) {
648  case 'browse':
649  $tr_success = LookedUpWord::NO_TR_LINK;
650  break;
651  case 'proofread':
652  // does not yet exist.
653  break;
654  case 'translate':
655  if($tr_success == LookedUpWord::SUCCESSFUL) {
656  $tr_success = LookedUpWord::NO_TR_LINK;
657  }
658  break;
659  case 'edit':
660  // no need to do anything
661  break;
662  }
663  return new LookedUpWord($code, $lang, $lookup_result, $tr_success, $tr_quality);
664  }
665  }
